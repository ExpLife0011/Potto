40  const InterfaceEntry* pEntries,
41  const PottoUuid& riid,
42  void** ppvObject)
43  {
44  PottoAssert(pThis != nullptr, "This pointer is null.");
45  PottoAssert(pEntries != nullptr, "Interface entry list is null.");
46 
47  if (pThis == nullptr || pEntries == nullptr)
48  return POTTO_E_INVALIDARG;
49 
50  if (ppvObject == nullptr)
51  return POTTO_E_POINTER;
52 
53  if (IsEqualUuid(IID_IUnknown, riid))
54  {
55  IUnknown* pUnk = (IUnknown*)((std::intptr_t)pThis + pEntries->offset);
56  pUnk->AddRef();
57  *ppvObject = pUnk;
58  return POTTO_E_OK;
59  }
60 
61  POTTO_ERROR error;
62 
63  for (;; pEntries++)
64  {
65  if (INTERFACE_INVALID_OFFSET == pEntries->offset)
66  {
67  error = POTTO_E_NOINTERFACE;
68  break;
69  }
70 
71  if (IsEqualUuid(pEntries->iid, riid))
72  {
73  IUnknown* pUnk = (IUnknown*)((std::intptr_t)pThis + pEntries->offset);
74  pUnk->AddRef();
75  *ppvObject = pUnk;
76  return POTTO_E_OK;
77  }
78  }
79 
80  *ppvObject = nullptr;
81 
82  return error;
